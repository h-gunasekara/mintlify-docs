# Response Structure

This guide explains the structure of responses from the Catalog API. All successful API calls return responses with a consistent format to make integration straightforward.

## Success Response Format

When your API request is successful, you'll receive a `200 OK` response with a JSON body containing:

```json
{
  "products": [ ... ],  // Array of product objects
  "meta": { ... }       // Metadata about the results
}
```

### The `products` Array

The `products` array contains a list of product objects matching your search and filter criteria. Each product has a comprehensive structure:

```json
{
  "id": 2458,
  "title": "Oversized Logo Hoodie",
  "description": "Iconic oversized silhouette hoodie featuring our signature logo print...",
  "vendor": "Balenciaga",
  "url": "https://www.balenciaga.com/en-us/oversized-logo-hoodie-black-600583TJVE11055.html",
  "is_available": true,
  "attributes": {
    "category": "Men > Hoodies & Sweatshirts",
    "color": "Black",
    "fabric": { "cotton": true, "polyester": true },
    "product_id": "60JVE55"
  },
  "images": [
    { "url": "https://assets.balenciaga.com/images/60JVE55.jpg", "alt": "Black oversized hoodie back" }
  ],
  "variants": [
    {
      "url": "https://www.balenciaga.com/en-us/oversized-logo-hoodie-black-60JVE55.html",
      "image": {
        "url": "https://assets.balenciaga.com/images/60JVE55.jpg"
      },
      "title": "Oversized Logo Hoodie - Black",
      "options": [
        {
          "size": "S",
          "price": {
            "value": 950,
            "currency": "USD",
            "display_value": 950
          },
          "is_in_stock": true
        },
        // More size options...
      ]
    }
  ],
  "price_currency": "USD",
  "price_display": 950
}
```

#### Product Object Fields

<ResponseField name="id" type="integer">
  Unique identifier for the product
</ResponseField>

<ResponseField name="title" type="string">
  Product title or name
</ResponseField>

<ResponseField name="description" type="string">
  Detailed product description
</ResponseField>

<ResponseField name="vendor" type="string">
  Brand or vendor name
</ResponseField>

<ResponseField name="url" type="string">
  URL to the product detail page
</ResponseField>

<ResponseField name="is_available" type="boolean">
  Whether the product is currently available
</ResponseField>

<ResponseField name="attributes" type="object">
  Object containing various product attributes:
  
  <Expandable title="Attribute Details">
    <ResponseField name="category" type="string">
      Product category hierarchy (e.g., "Men > Hoodies & Sweatshirts")
    </ResponseField>
    
    <ResponseField name="color" type="string">
      Product color name
    </ResponseField>
    
    <ResponseField name="fabric" type="object">
      Object with fabric types as keys and boolean values
    </ResponseField>
    
    <ResponseField name="product_id" type="string">
      Vendor's product identifier
    </ResponseField>
    
    <ResponseField name="..." type="varies">
      Other attributes may be present depending on the product
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="images" type="array">
  Array of image objects:
  
  <Expandable title="Image Object">
    <ResponseField name="url" type="string">
      URL to the image file
    </ResponseField>
    
    <ResponseField name="alt" type="string">
      Alt text for the image (may be null)
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="variants" type="array">
  Array of product variants:
  
  <Expandable title="Variant Object">
    <ResponseField name="url" type="string">
      URL to the specific variant
    </ResponseField>
    
    <ResponseField name="image" type="object">
      Image specific to this variant
    </ResponseField>
    
    <ResponseField name="title" type="string">
      Title of this variant
    </ResponseField>
    
    <ResponseField name="options" type="array">
      Array of option objects (typically sizes):
      
      <Expandable title="Option Object">
        <ResponseField name="size" type="string">
          Size identifier (e.g., "S", "M", "L")
        </ResponseField>
        
        <ResponseField name="price" type="object">
          Price information for this option:
          
          <Expandable title="Price Object">
            <ResponseField name="value" type="number">
              Numeric price value
            </ResponseField>
            
            <ResponseField name="currency" type="string">
              Currency code (e.g., "USD")
            </ResponseField>
            
            <ResponseField name="display_value" type="number">
              Formatted price for display
            </ResponseField>
          </Expandable>
        </ResponseField>
        
        <ResponseField name="is_in_stock" type="boolean">
          Whether this option is currently in stock
        </ResponseField>
      </Expandable>
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="price_currency" type="string">
  Currency code for the product's price
</ResponseField>

<ResponseField name="price_display" type="number">
  Default display price for the product
</ResponseField>

### The `meta` Object

The `meta` object contains metadata about the results:

```json
"meta": {
  "totalItems": 50,
  "totalPages": 15,
  "currentPage": 1,
  "pageSize": 10
}
```

<ResponseField name="totalItems" type="integer">
  Total number of products matching the search and filter criteria across all pages. <strong>Note: This value is capped at 50 regardless of the actual total.</strong>
</ResponseField>

<ResponseField name="totalPages" type="integer">
  Total number of pages available based on `totalItems` and `pageSize`.
</ResponseField>

<ResponseField name="currentPage" type="integer">
  The current page number (matches the `page` parameter from the request).
</ResponseField>

<ResponseField name="pageSize" type="integer">
  Number of items per page.
</ResponseField>

## Empty Results

If your query doesn't match any products, you'll still get a successful response (`200 OK`), but with an empty products array:

```json
{
  "products": [],
  "meta": {
    "totalItems": 0,
    "totalPages": 0,
    "currentPage": 1,
    "pageSize": 10
  }
}
```

## Error Responses

When an error occurs, the API returns an appropriate HTTP status code and a JSON body with error details:

### Authentication Error (401 Unauthorized)

```json
{
  "error": "Unauthorized",
  "message": "Missing or invalid API key"
}
```

### Validation Error (400 Bad Request)

```json
{
  "error": "Bad Request",
  "message": "Invalid request parameters",
  "details": {
    "price_min": "Must be a positive number"
  }
}
```

### Rate Limiting Error (429 Too Many Requests)

```json
{
  "error": "Rate Limit Exceeded",
  "message": "You have exceeded your rate limit",
  "retry_after": 60
}
```

### Server Error (500 Internal Server Error)

```json
{
  "error": "Internal Server Error",
  "message": "An unexpected error occurred"
}
```

## Working with Responses

### Handling Pagination

Use the `meta` object to implement pagination controls:

```javascript
const totalPages = response.meta.totalPages;
const currentPage = response.meta.currentPage;

// Check if there are more pages
const hasNextPage = currentPage < totalPages;
const hasPrevPage = currentPage > 1;

// Calculate next/prev page numbers
const nextPage = hasNextPage ? currentPage + 1 : null;
const prevPage = hasPrevPage ? currentPage - 1 : null;
```

### Displaying Products

The comprehensive structure lets you display rich product information:

```javascript
// Example: Rendering basic product info
response.products.forEach(product => {
  console.log(`${product.title} - $${product.price_display} ${product.price_currency}`);
  console.log(`  Vendor: ${product.vendor}`);
  console.log(`  Color: ${product.attributes.color}`);
  console.log(`  Available: ${product.is_available ? 'Yes' : 'No'}`);
  
  // Get the main image URL (if available)
  const mainImageUrl = product.images.length > 0 ? product.images[0].url : 'no-image.jpg';
  
  // Get available sizes
  const availableSizes = product.variants[0].options
    .filter(option => option.is_in_stock)
    .map(option => option.size)
    .join(', ');
  
  console.log(`  Available sizes: ${availableSizes}`);
});
```

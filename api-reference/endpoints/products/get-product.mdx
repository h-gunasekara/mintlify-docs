---
title: "Get Product"
api: "GET https://catalogai.vercel.app/api/products/{id}"
description: "Get detailed information about a specific product"
---

Retrieve comprehensive details about a specific product by its unique identifier.

## Request

<ParamField header="x-api-key" type="string" required>
  Your API key for authentication
</ParamField>

### Path Parameters

<ParamField path="id" type="string" required>
  The unique identifier of the product
</ParamField>

## Response

<ResponseField name="product" type="object">
  Detailed product information
  
  <Expandable title="Product Properties">
    <ResponseField name="id" type="string">
      Unique product identifier
    </ResponseField>
    
    <ResponseField name="title" type="string">
      Product title
    </ResponseField>
    
    <ResponseField name="description" type="string">
      Detailed product description
    </ResponseField>
    
    <ResponseField name="vendor" type="string">
      Brand or vendor name
    </ResponseField>
    
    <ResponseField name="price" type="number">
      Price in cents
    </ResponseField>
    
    <ResponseField name="currency" type="string">
      Currency code (e.g., "USD")
    </ResponseField>
    
    <ResponseField name="is_available" type="boolean">
      Product availability status
    </ResponseField>
    
    <ResponseField name="image_url" type="string">
      Primary product image URL
    </ResponseField>
    
    <ResponseField name="product_url" type="string">
      Link to product page
    </ResponseField>
    
    <ResponseField name="created_at" type="string">
      Product creation timestamp (ISO 8601)
    </ResponseField>
    
    <ResponseField name="updated_at" type="string">
      Last update timestamp (ISO 8601)
    </ResponseField>
    
    <ResponseField name="attributes" type="object">
      Comprehensive product attributes
      
      <Expandable title="Available Attributes">
        <ResponseField name="color" type="string">
          Primary product color
        </ResponseField>
        
        <ResponseField name="material" type="string">
          Primary material
        </ResponseField>
        
        <ResponseField name="category" type="string">
          Product category hierarchy
        </ResponseField>
        
        <ResponseField name="gender" type="string">
          Target gender
        </ResponseField>
        
        <ResponseField name="season" type="string">
          Seasonal appropriateness
        </ResponseField>
        
        <ResponseField name="occasion" type="string">
          Suitable occasion
        </ResponseField>
        
        <ResponseField name="size_range" type="array">
          Available sizes
        </ResponseField>
      </Expandable>
    </ResponseField>
  </Expandable>
</ResponseField>

<RequestExample>

```bash cURL
curl -X GET https://catalogai.vercel.app/api/products/12345 \
  -H "x-api-key: YOUR_API_KEY"
```

```javascript JavaScript
const response = await fetch('https://catalogai.vercel.app/api/products/12345', {
  headers: {
    'x-api-key': 'YOUR_API_KEY'
  }
});

const data = await response.json();
```

```python Python
import requests

response = requests.get(
    'https://catalogai.vercel.app/api/products/12345',
    headers={'x-api-key': 'YOUR_API_KEY'}
)

data = response.json()
```

</RequestExample>
